if __name__ == '__main__':
    n = int(input())
    student_marks = {}
    for _ in range(n):
        name, *line = input().split()
        scores = list(map(float, line))
        student_marks[name] = scores
    query_name = input()
    list_score=student_marks[query_name]
    
    perc =0
    for i in list_score:
        perc+=i
        
    print("{0:.2f}".format(perc/len(list_score)))
